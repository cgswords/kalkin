\usepackage{amsmath, listings, amsthm, amssymb, xspace}

%%%%%%%%%%%%%%%%%%%%%%%%
%% DEFINITION HELPERS %%
%%%%%%%%%%%%%%%%%%%%%%%%

\newcommand{\alt}{~~|~~}
\newcommand{\comp}[1]{\llbracket #1 \rrbracket}

\newcommand{\inlinexp}[1]{
{\footnotesize
 \[\begin{array}{l}
 #1
 \end{array}\]}}

\newcommand{\inlinexpa}[2]{
{\footnotesize
 \[\begin{array}{#1}
 #2
 \end{array}\]}}

%%%%%%%%%%%%%%%%%%%%%%%
%% Expression Macros %%
%%%%%%%%%%%%%%%%%%%%%%%

%% let

\newcommand{\letbinde} [2] {\mathsf{let}~\binde{#1}{#2}}
\newcommand{\binde}    [2] {#1:=#2}

%% if
\newcommand{\ife}      [3] {\ifop~#1~#2~#3}
\newcommand{\ifop}         {\mathsf{if}}

%% callables
\newcommand{\lambdae}  [2] {\lambda~\namet_0 \ldots \namet_#1~#2}
\newcommand{\fundefo}  [2] {\namet_{name}~\namet_0 \ldots \namet_#1~#2}

%% call
\newcommand{\calle}    [2] {#1~\exprt_0 \ldots \exprt_#2}

%% values
\newcommand{\truev}        {\mathsf{true}}
\newcommand{\falsev}       {\mathsf{false}}

%% environments
\newcommand{\stackenv}     {\sigma}
\newcommand{\heapenv}      {\mu}
\newcommand{\namespace}    {\nu}

%% terms
\newcommand{\addrt}        {a}
\newcommand{\exprt}        {e}
\newcommand{\funt}         {f}
\newcommand{\namet}        {x}
\newcommand{\objt}         {o}
